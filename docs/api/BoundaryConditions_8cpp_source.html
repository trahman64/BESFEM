221void BoundaryConditions::SetupPinnedDOF(mfem::ParFiniteElementSpace &fespace)
222{
223 myid = mfem::Mpi::WorldRank();
224
225 if (E_L2G.Size() == 0)
226 parallelMesh.GetGlobalElementIndices(E_L2G);
227
228 double cand_dist = 0.0;
229 int local_gVpp = SelectCenterPin(0.99, cand_dist);
230
231 const double posInf = std::numeric_limits<double>::infinity();
232
233 if (local_gVpp < 0)
234 {
235 cand_dist = posInf;
236 }
237
238 double local_d = cand_dist;
239 double best_d = 0.0;
240 MPI_Allreduce(&local_d, &best_d, 1, MPI_DOUBLE, MPI_MIN, MPI_COMM_WORLD); // find the lowest global distance from center value
241
242 const double d_tol = 1e-14;
243
244 int filtered_vid = (std::abs(cand_dist - best_d) <= d_tol && local_gVpp >= 0) ? local_gVpp: std::numeric_limits<int>::max();
245
246 int gVpp = 0;
247 MPI_Allreduce(&filtered_vid, &gVpp, 1, MPI_INT, MPI_MIN, MPI_COMM_WORLD); // find the smallest gVpp after finding the smallest distance
248
249 if (gVpp == std::numeric_limits<int>::max())
250 {
251 if (myid == 0)
252 std::cerr << "[SetupPinnedDOF] Failed to elect a global pinned vertex.\n";
253 return;
254 }
255
256 int lVpp = -1;
257 pin = false;
258 rkpp = -1;
259
260 for (int ei = 0; ei < geometry.nE; ei++)
261 {
262 int gei = E_L2G[ei];
263 globalMesh.GetElementVertices(gei, gVTX);
264 parallelMesh.GetElementVertices(ei, VTX);
265
266 for (int vi = 0; vi < geometry.nC; vi++)
267 {
268 if (gVTX[vi] == gVpp)
269 {
270 lVpp = VTX[vi]; // local vertex index on this rank
271 pin = true;
272 rkpp = myid;
273 }
274 }
275 }
276
277 if (pin)
278 {
279 mfem::Array<int> vdofs;
280 fespace.GetVertexVDofs(lVpp, vdofs);
281
282 ess_tdof_marker.SetSize(fespace.GetTrueVSize());
283 ess_tdof_marker = 0;
284
285 int ldof = vdofs[0];
286 if (ldof < 0)
287 ldof = -1 - ldof;
288
289 int ltdof = fespace.GetLocalTDofNumber(ldof);
290 if (ltdof >= 0)
291 ess_tdof_marker[ltdof] = 1;
292
293 fespace.MarkerToList(ess_tdof_marker, ess_tdof_list);

295 std::cout << "Rank " << myid

306int BoundaryConditions::SelectCenterPin(double threshold, double &out_dist2)
307{
308 myid = mfem::Mpi::WorldRank();
309
310 if (!domain_parameters.pse)
311 {
312 if (myid == 0)
313 std::cerr << "[SelectPin] pse not initialized!\n";
314 out_dist2 = std::numeric_limits<double>::infinity();
315 return -1;
316 }
317
318 const mfem::ParGridFunction &pse_field = *domain_parameters.pse;
319
320 int nE = parallelMesh.GetNE();
321 int dim = parallelMesh.Dimension();
322
323 mfem::Vector minv(dim), maxv(dim);
324 globalMesh.GetBoundingBox(minv, maxv);
325
326 mfem::Vector center(dim);
327 for (int d = 0; d < dim; d++)
328 center[d] = 0.5 * (minv[d] + maxv[d]); // find the center
329
330 if (E_L2G.Size() == 0)
331 parallelMesh.GetGlobalElementIndices(E_L2G);
332
333 mfem::Vector ec(dim);
334 mfem::Array<int> VTX;
335 mfem::Array<int> vdofs;
336
337 double best_dist2 = std::numeric_limits<double>::infinity(); // first make the best distance positive inf
338 int best_ei = -1; // first make the best element -1
339
340 const double z_tol = 1e-12; // tolerance for comparing z to min/max
341
342 // loop over vertices and find what is inside electrolyte
343 for (int ei = 0; ei < nE; ei++)
344 {
345 parallelMesh.GetElementVertices(ei, VTX);
346 int nV = VTX.Size();
347
348 bool inside = false;
349
350 if (dim == 2)
351 {
352 inside = false;
353 for (int vi = 0; vi < nV; vi++)
354 {
355 if (pse_field(VTX[vi]) >= threshold)
356 {
357 inside = true;
358 break;
359 }
360 }
361 }
362 else if (dim == 3)
363 {
364 inside = false;
365 for (int vi = 0; vi < nV; vi++)
366 {
367 parfespace.GetVertexDofs(VTX[vi], vdofs);
368 double val = pse_field(vdofs[0]);
369 if (val >= threshold)
370 {
371 inside = true;
372 break;
373 }
374 }
375 }
376
377 if (!inside) continue;
378
379 int gei = E_L2G[ei];
380 globalMesh.GetElementCenter(gei, ec);
381
382 if (dim == 3)
383 {
384 double zc = ec[2];
385 double zmin = minv[2];
386 double zmax = maxv[2];
387
388 if (std::abs(zc - zmin) < z_tol || std::abs(zc - zmax) < z_tol)
389 {
390 continue;
391 }
392 }
393
394 double dx = ec[0] - center[0]; // distance in x from global center
395 double dy = (dim > 1) ? (ec[1] - center[1]) : 0.0; // distance in y from global center
396 double dist2 = dx*dx + dy*dy;
397
398 // pick the element closest to the center in the xy-plane;
399 // ties are automatically broken by "first one encountered"
400 if (dist2 < best_dist2)
401 {
402 best_dist2 = dist2;
403 best_ei = ei;
404 }
405
406 }
407
408 // all elements not the best one are -1
409 if (best_ei < 0)
410 {
411 out_dist2 = std::numeric_limits<double>::infinity();
412 return -1;
413 }
414
415 // need to find which vertex to pin
416 parallelMesh.GetElementVertices(best_ei, VTX);
417 int nV = VTX.Size();
418 int gei = E_L2G[best_ei];
419
420 mfem::Array<int> gVTX(nV);
421 globalMesh.GetElementVertices(gei, gVTX);
422
423 int chosen_gv = gVTX[0]; // default
424
425 if (dim == 3)
426 {
427 double zmin = minv[2];
428 double zmax = maxv[2];
429 const double z_tol = 1e-12; // same tolerance as before
430
431 chosen_gv = -1;
432
433 // Prefer a vertex whose z is strictly between zmin and zmax
434 for (int vi = 0; vi < nV; vi++)
435 {
436 const double *X = globalMesh.GetVertex(gVTX[vi]); // [x,y,z]
437 double z = X[2];
438
439 if (std::abs(z - zmin) >= z_tol && std::abs(z - zmax) >= z_tol)
440 {
441 chosen_gv = gVTX[vi];
442 break; // first such vertex wins
443 }
444 }
445
446 // if all vertices are on zmin or zmax (should be rare), just pick the first one so we still have something to pin
447 if (chosen_gv < 0)
448 chosen_gv = gVTX[0];
449 }
450 else
451 {
452 // 2D: z doesn't matter
453 chosen_gv = gVTX[0];
454 }
455
456 out_dist2 = best_dist2;
457 return chosen_gv; // GLOBAL vertex index candidate on this rank
458
459}

462int BoundaryConditions::SelectFirstPin(double threshold, double &out_dist2)
463{
464 myid = mfem::Mpi::WorldRank();
465
466 if (!domain_parameters.pse)
467 {
468 if (myid == 0)
469 std::cerr << "[SelectFirstPin] pse not initialized!\n";
470 out_dist2 = std::numeric_limits<double>::infinity();
471 return -1;
472 }
473
474 const mfem::ParGridFunction &pse_field = *domain_parameters.pse;
475
476 int nE = parallelMesh.GetNE();
477 int dim = parallelMesh.Dimension();
478
479 // ------------------------------
480 // Bounding box for boundary check
481 // ------------------------------
482 mfem::Vector minv(dim), maxv(dim);
483 globalMesh.GetBoundingBox(minv, maxv);
484
485 const double tol = 1e-12;
486 // const double tol = Constants::dh;
487
488 if (E_L2G.Size() == 0)
489 parallelMesh.GetGlobalElementIndices(E_L2G);
490
491 mfem::Vector ec(dim);
492 mfem::Array<int> VTX, gVTX, vdofs;
493
494 // ------------------------------
495 // Loop over elements
496 // ------------------------------
497 for (int ei = 0; ei < nE; ei++)
498 {
499 parallelMesh.GetElementVertices(ei, VTX);
500 int nV = VTX.Size();
501
502 // --- Check pse to see if inside electrolyte ---
503 bool inside = false;
504
505 for (int vi = 0; vi < nV; vi++)
506 {
507 parfespace.GetVertexDofs(VTX[vi], vdofs);
508 double val = pse_field(vdofs[0]);
509
510 if (val >= threshold)
511 {
512 inside = true;
513 break;
514 }
515 }
516
517 if (!inside) continue;
518
519 // --- Element center in global coords ---
520 int gei = E_L2G[ei];
521 globalMesh.GetElementCenter(gei, ec);
522
523 double xc = ec[0];
524 double yc = (dim > 1 ? ec[1] : 0.0);
525 double zc = (dim > 2 ? ec[2] : 0.0);
526
527 // Compute local element cell sizes dh_x, dh_y, dh_z from its vertices
528 double ex_min = 1e300, ex_max = -1e300;
529 double ey_min = 1e300, ey_max = -1e300;
530 double ez_min = 1e300, ez_max = -1e300;
531
532 globalMesh.GetElementVertices(gei, gVTX);
533 for (int vi = 0; vi < gVTX.Size(); vi++)
534 {
535 const double *Xv = globalMesh.GetVertex(gVTX[vi]);
536 ex_min = std::min(ex_min, Xv[0]);
537 ex_max = std::max(ex_max, Xv[0]);
538 if (dim > 1) {
539 ey_min = std::min(ey_min, Xv[1]);
540 ey_max = std::max(ey_max, Xv[1]);
541 }
542 if (dim > 2) {
543 ez_min = std::min(ez_min, Xv[2]);
544 ez_max = std::max(ez_max, Xv[2]);
545 }
546 }
547
548 double dh_x = ex_max - ex_min;
549 double dh_y = (dim > 1 ? ey_max - ey_min : 1.0);
550 double dh_z = (dim > 2 ? ez_max - ez_min : 1.0);
551
552 // Reject if center within certain number of elements in x and y direction
553 // if the element is too close to and x or y direction, then I keep looking for another element
554 bool boundary_check = ((xc - minv[0]) < 4.0 * dh_x) || ((maxv[0] - xc) < 4.0 * dh_x) ||
555 (dim > 1 && ((yc - minv[1]) < 4.0 * dh_y || (maxv[1] - yc) < 4.0 * dh_y));
556
557 if (boundary_check) // if the element ends up not being too close, then do not continue and keep that for the pin
558 continue;
559
560
561 // ------------------------------
562 // FIRST valid element found!
563 // Now select the first interior vertex of this element
564 // ------------------------------
565 globalMesh.GetElementVertices(gei, gVTX);
566
567 // pick vertex that is interior in ALL dims
568 for (int vi = 0; vi < nV; vi++)
569 {
570 const double *X = globalMesh.GetVertex(gVTX[vi]);
571 double x = X[0];
572 double y = (dim > 1 ? X[1] : 0.0);
573 double z = (dim > 2 ? X[2] : 0.0);
574
575 bool vtx_on_boundary =
576 (std::abs(x - minv[0]) < tol) || (std::abs(x - maxv[0]) < tol) ||
577 (dim > 1 && (std::abs(y - minv[1]) < tol || std::abs(y - maxv[1]) < tol)) ||
578 (dim > 2 && (std::abs(z - minv[2]) < tol || std::abs(z - maxv[2]) < tol));
579
580 if (!vtx_on_boundary)
581 {
582 out_dist2 = 0.0; // For MPI comparability
583 return gVTX[vi]; // GLOBAL vertex index
584 }
585 }
586
587 // If all vertices lie on boundary, skip element
588 // (rare but possible with structured meshes)
589 }
590
591 // No interior element found
592 out_dist2 = std::numeric_limits<double>::infinity();
593 return -1;
