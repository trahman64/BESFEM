19static inline void GlobalMinMax(const mfem::ParGridFunction& gf,
20 double& gmin, double& gmax,
21 MPI_Comm comm = MPI_COMM_WORLD)
22{
23 double lmin = std::numeric_limits<double>::infinity();
24 double lmax = -std::numeric_limits<double>::infinity();
25 for (int i = 0; i < gf.Size(); ++i) {
26 const double v = gf(i);
27 if (v < lmin) lmin = v;
28 if (v > lmax) lmax = v;
29 }
30 MPI_Allreduce(&lmin, &gmin, 1, MPI_DOUBLE, MPI_MIN, comm);
31 MPI_Allreduce(&lmax, &gmax, 1, MPI_DOUBLE, MPI_MAX, comm);
32}

341 auto ComputeGradMagnitude = [&](const mfem::ParGridFunction &phase_in,
342 mfem::ParGridFunction &AvP_out)
343 {
344 const int dim = pmesh->Dimension();
345 mfem::ParGridFunction dphase(fespace.get());
346
347 AvP_out = 0.0;
348 for (int d = 0; d < dim; ++d)
349 {
350 dphase = 0.0;
351 mfem::ParGridFunction phase_tmp(phase_in);
352 phase_tmp.GetDerivative(1, d, dphase);
353
354 for (int vi = 0; vi < nV; ++vi)
355 {
356 const double v = dphase(vi);
357 AvP_out(vi) += v * v;
358 }
359 }
360
361 for (int vi = 0; vi < nV; ++vi)
362 {
363 AvP_out(vi) = std::sqrt(AvP_out(vi));
364 }
365 };

369 auto BuildPairInterface = [&](mfem::ParGridFunction &out,
370 const mfem::ParGridFunction &psa,
371 const mfem::ParGridFunction &psb,
372 const mfem::ParGridFunction &AvPa,
373 const mfem::ParGridFunction &AvPb)
374 {
375 out = psa; // psa
376 out *= AvPb; // psa * AvPb
377
378 mfem::ParGridFunction tmp(fespace.get());
379 tmp = psb; // psb
380 tmp *= AvPa; // psb * AvPa
381 out += tmp; // psa*AvPb + psb*AvPa
382
383 mfem::ParGridFunction overlap(fespace.get());
384 overlap = psa;
385 overlap *= psb; // psa * psb
386
387 out *= overlap;
388 out *= 4.0;
389
390 for (int vi = 0; vi < out.Size(); ++vi)
391 {
392 if (out(vi) > 9000.0)
393 {
394 out(vi) = 1.4e4;
395 }
396 }

408 auto ComputeWeight = [&](mfem::ParGridFunction &weight_out,
409 const mfem::ParGridFunction &num_in,
410 const mfem::ParGridFunction *mask_in = nullptr)
411 {
412 weight_out = 0.0;
413
414 const double beta = 0.8;
415 const double eps = 1e-30;
416
417 for (int vi = 0; vi < nV; ++vi)
418 {
419 const double num = num_in(vi);
420 const double den = (*denom)(vi);
421
422 double ratio = 0.0;
423 if (den > eps)
424 {
425 ratio = num / den;
426 if (ratio < 0.0) ratio = 0.0;
427 }
428
429 weight_out(vi) = std::pow(ratio, beta);
430 }
431
432 if (mask_in) { weight_out *= *mask_in; }
433 };

435 auto BuildPsiPairs = [&](mfem::ParGridFunction &out,
436 const mfem::ParGridFunction &psa,
437 const mfem::ParGridFunction &psb)
438 {
439 out = psa;
440 out += psb;

443 {

445 }

450 // Solving
451 // -------------------------------------------------
452
453 // Building AvPs

455 ComputeGradMagnitude(*pse, *AvE); // electrolyte
456
457 for (int k = 0; k < (int)ps.size(); ++k)
458 {

460 }
461

464 {

466 {

468 BuildPsiPairs(*psi_Pairs[j][k], *ps[j], *ps[k]);
469 }
470 }
471
472 // Building AvEs
473 for (int k = 0; k < (int)ps.size(); ++k)
474 {
475 BuildElectrolyteInterface(*AvEs[k], *pse, *AvPs[k]);
476 }
477
478 // Building denominator for weights
479 *denom = 0.0;
480 for (int j = 0; j < (int)ps.size(); ++j)
481 {
482 for (int k = j + 1; k < (int)ps.size(); ++k)
483 {
484 *denom += *AvP_Pairs[j][k];
485 }
486 }
487 for (int k = 0; k < (int)ps.size(); ++k)
488 {
489 *denom += *AvEs[k];
490 }
491
492 // Building weights
493 for (int k = 0; k < (int)ps.size(); ++k)
494 {
495 ComputeWeight(*WeightEs[k], *AvEs[k]);
496 }
497
498 for (int j = 0; j < (int)ps.size(); ++j)
499 {
500 for (int k = j + 1; k < (int)ps.size(); ++k)
501 {
502 ComputeWeight(*WeightPairs[j][k], *AvP_Pairs[j][k], psi_Pairs[j][k].get());
503 }
504 }
505
506 // ---- GLOBAL checks for psi -------------------------------------------
507 double psi_min = 0.0, psi_max = 0.0;
508 GlobalMinMax(*psi, psi_min, psi_max);
509
510 // Basic bounds check
511 if (mfem::Mpi::WorldRank() == 0) {std::cout << "[Psi Check] min = " << psi_min

529void Domain_Parameters::CalculateTotals(const mfem::ParGridFunction& grid_function, const mfem::Vector& element_volumes, double& local_total, double& global_total) {
530 local_total = 0.0;
531
532 // Average value for each element
533 mfem::Vector element_avg_values(nE);
534
535 for (int ei = 0; ei < pmesh->GetNE(); ei++) {
536
537 mfem::Array<double> vertex_values;
538 grid_function.GetNodalValues(ei, vertex_values);
539
540 double avg_value = 0.0;
541 for (int vt = 0; vt < vertex_values.Size(); vt++) {
542 avg_value += vertex_values[vt];
543 }
544 avg_value /= vertex_values.Size();
545
546 element_avg_values(ei) = avg_value;
547
548 // Accumulate the weighted value for the current element
549 local_total += avg_value * element_volumes(ei);
550 }
551
552 // Perform global MPI reduction to sum values across all processes
553 MPI_Allreduce(&local_total, &global_total, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
554}

591void Domain_Parameters::CalculateTargetCurrent(double total_psi, double &global_total, sim::MaterialType material) {
592
593 double rho = MaterialProperties::SiteDensity(material);
594
595 // Compute target current based on total Psi, rho, Cr, and constants
