19static inline void GlobalMinMax(const mfem::ParGridFunction& gf,
20 double& gmin, double& gmax,
21 MPI_Comm comm = MPI_COMM_WORLD)
22{
23 double lmin = std::numeric_limits<double>::infinity();
24 double lmax = -std::numeric_limits<double>::infinity();
25 for (int i = 0; i < gf.Size(); ++i) {
26 const double v = gf(i);
27 if (v < lmin) lmin = v;
28 if (v > lmax) lmax = v;
29 }
30 MPI_Allreduce(&lmin, &gmin, 1, MPI_DOUBLE, MPI_MIN, comm);
31 MPI_Allreduce(&lmax, &gmax, 1, MPI_DOUBLE, MPI_MAX, comm);
32}

534void Domain_Parameters::CalculateTotals(const mfem::ParGridFunction& grid_function, const mfem::Vector& element_volumes, double& local_total, double& global_total) {
535 local_total = 0.0;
536
537 // Average value for each element
538 mfem::Vector element_avg_values(nE);
539
540 for (int ei = 0; ei < pmesh->GetNE(); ei++) {
541 mfem::Array<double> vertex_values(nC);
542 grid_function.GetNodalValues(ei, vertex_values);
543
544 // Compute the average value over all corners of the element
545 double avg_value = 0.0;
546 for (int vt = 0; vt < nC; vt++) {
547 avg_value += vertex_values[vt];
548 }
549 avg_value /= nC;
550
551 element_avg_values(ei) = avg_value;
552
553 // Accumulate the weighted value for the current element
554 local_total += avg_value * element_volumes(ei);
555 }
556
557 // Perform global MPI reduction to sum values across all processes
558 MPI_Allreduce(&local_total, &global_total, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
559}

613void Domain_Parameters::CalculateTargetCurrent(double total_psi, double &global_total, sim::MaterialType material) {
614
615 double rho = MaterialProperties::SiteDensity(material);
616
617 // Compute target current based on total Psi, rho, Cr, and constants
618 trgI = total_psi * rho * (0.95 - 0.3) / (3600.0 / cfg.Cr); // bounds of cathode
619
