44double Reaction::GetTableValues(double cn, const mfem::Vector &ticks, const mfem::Vector &data)
45{
46 if (cn < 1.0e-6) cn = 1.0e-6;
47 if (cn > 0.999999) cn = 0.999999;
48
49 int idx = std::floor(cn / 0.01);
50 if (idx < 0) idx = 0;
51 if (idx > 99) idx = 99;
52
53 return data(idx) + (cn - ticks(idx)) / 0.01 * (data(idx + 1) - data(idx));

166void Reaction::TotalReactionCurrent(mfem::ParGridFunction &Rx, double &global_current)
167{
168 local_current = 0.0;
169 mfem::Array<double> VtxVal(nC);
170 mfem::Vector EAvg(nE);
171
172 for (int ei = 0; ei < nE; ++ei) {
173 Rx.GetNodalValues(ei, VtxVal);
174 double sum = std::accumulate(VtxVal.begin(), VtxVal.end(), 0.0);
175 EAvg(ei) = sum / nC;
176 local_current += EAvg(ei) * EVol(ei);
177 }
178
179 MPI_Allreduce(&local_current, &global_current, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
