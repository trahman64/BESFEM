142 namespace fs = std::filesystem;
143
144 auto now = std::chrono::system_clock::now();
145 std::time_t now_c = std::chrono::system_clock::to_time_t(now);
146 std::tm tm{};
147
148 #if defined(_WIN32)
149 localtime_s(&tm, &now_c);
150 #else
151 localtime_r(&now_c, &tm);
152 #endif
153
154 std::ostringstream ts;
155 ts << std::put_time(&tm, "%Y%m%d_%H%M%S");
156 std::string mesh_name = fs::path(mesh_file).stem().string();
157
158 std::ostringstream od;
159 od << "../outputs/Results/" << ts.str();
160 // << "__nsteps=" << num_steps
161 // << "__mesh=" << mesh_name;
162
163 return od.str();
164 }
