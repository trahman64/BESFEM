48void Utils::CalculateReactionInfx(mfem::ParGridFunction &Rx, double &xCrnt)
49{
50 xCrnt = 0.0;
51
52 mfem::Vector Rmin, Rmax;
53 pmesh_->GetBoundingBox(Rmin, Rmax);
54
55 double Lw = (Rmax(1) - Rmin(1));
56 if (pmesh_->Dimension() == 3)
57 Lw *= (Rmax(2) - Rmin(2));
58
59 for (int ei = 0; ei < nE_; ei++)
60 {
61 Rx.GetNodalValues(ei, VtxVal_);
62 double sum = 0;
63 for (int i = 0; i < nC_; i++) sum += VtxVal_[i];
64
65 EAvg_(ei) = sum / nC_;
66 xCrnt += EAvg_(ei) * EVol_(ei);
67 }
68
69 MPI_Allreduce(&xCrnt, &geCrnt_, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
70 infx_ = geCrnt_ / Lw;

73void Utils::CalculateGlobalError(mfem::ParGridFunction &px0, mfem::ParGridFunction &potential, mfem::ParGridFunction &psx, double &globalerror, double gtPsx)
74{
75 TmpF_ = px0;
76 TmpF_ -= potential;
77 TmpF_ *= TmpF_;
78 TmpF_ *= psx;
79
80 double local_sum = TmpF_.Sum();
81 MPI_Allreduce(&local_sum, &globalerror, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);

83 globalerror /= gtPsx;

179void Utils::SaveSimulationSnapshotMulti(int t, const std::string &outdir, Initialize_Geometry &geometry,
180 Domain_Parameters &domain_parameters, const std::vector<mfem::ParGridFunction*> &particle_cn,
181 std::vector<std::unique_ptr<mfem::ParGridFunction>> &particle_out, int save_interval)
182
183 {
184 if (t % save_interval != 0) return;
185
186 const int np = static_cast<int>(particle_cn.size());
187
188 std::ostringstream step;
189 step << "_" << std::setw(5) << std::setfill('0') << t;
190 const std::string suff = step.str();
191
192 if (t == 0)
193 {
194 geometry.parallelMesh->SaveAsOne((outdir + "/pmesh").c_str());
195 domain_parameters.psi->SaveAsOne((outdir + "/psi").c_str());
196 domain_parameters.pse->SaveAsOne((outdir + "/pse").c_str());
197 }
198
199 // Save each particle concentration and masked version
200 for (int k = 0; k < np; ++k)
201 {
202 std::ostringstream raw_name, masked_name;
203 raw_name << outdir << "/CnC_" << (k + 1) << suff;
204 masked_name << outdir << "/C" << (k + 1) << "_out" << suff;
205
206 particle_cn[k]->SaveAsOne(raw_name.str().c_str());
207
208 *particle_out[k] = *particle_cn[k];
209 *particle_out[k] *= *domain_parameters.ps[k];
210 particle_out[k]->SaveAsOne(masked_name.str().c_str());
211 }
212
213 // Build union mask and denominator
214 mfem::ParGridFunction psi_union(geometry.parfespace.get());
215 mfem::ParGridFunction denom(geometry.parfespace.get());
216 mfem::ParGridFunction CnP_total(geometry.parfespace.get());
217
218 psi_union = 0.0;
219 denom = 0.0;
220 CnP_total = 0.0;
221
222 for (int k = 0; k < np; ++k)
223 {
224 psi_union += *domain_parameters.ps[k];
225 denom += *domain_parameters.ps[k];
226 }
227
228 for (int i = 0; i < psi_union.Size(); ++i)
229 {
230 psi_union(i) = std::min(1.0, psi_union(i));
231 }
232
233 const double eps = 1e-30;
234
235 for (int i = 0; i < denom.Size(); ++i)
236 {
237 const double d = denom(i);
238
239 if (d > eps)
240 {
241 double num = 0.0;
242 for (int k = 0; k < np; ++k)
243 {
244 num += (*domain_parameters.ps[k])(i) * (*particle_cn[k])(i);
245 }
246 CnP_total(i) = num / (d + eps);
247 }
248 else
249 {
250 CnP_total(i) = 0.0;
251 }
252 }
253
254 CnP_total *= psi_union;
255 CnP_total.SaveAsOne((outdir + "/CnP_total" + suff).c_str());
256 }
