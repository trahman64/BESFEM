385void Pairs(SimulationState& state, Initialize_Geometry& geometry, Domain_Parameters& domain_parameters, int j, std::vector<ConcentrationBase::PairCoupling>& pair_terms, int np, int t)
386{
387 for (int k = 0; k < np; ++k)
388 {
389 if (j == k) { continue; }
390
391 const int a = std::min(j, k);
392 const int b = std::max(j, k);
393
394 ConcentrationBase::PairCoupling pair;
395 pair.sum_part = state.sum_pairs[a][b].get();
396 pair.weight = domain_parameters.WeightPairs[a][b].get();
397 pair.grad_psi = domain_parameters.AvP_Pairs[a][b].get();
398
399 if (j < k)
400 {
401 pair.mu_self = state.mu_pair_a[a][b].get();
402 pair.mu_nbr = state.mu_pair_b[a][b].get();
403 }
404 else
405 {
406 pair.mu_self = state.mu_pair_b[a][b].get();
407 pair.mu_nbr = state.mu_pair_a[a][b].get();
408 }
409
410 pair_terms.push_back(pair);
411
412 if (mfem::Mpi::WorldRank() == 0 && t == 1)
413 {
414 std::cout << "[DEBUG] Pair (j,k) = (" << j << "," << k << ")";
415 std::cout << std::endl;
416 }
417
418 }
