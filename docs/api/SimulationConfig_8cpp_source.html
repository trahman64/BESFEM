14static std::string Trim(const std::string& s)
15{
16 const auto first = s.find_first_not_of(" \t\r\n");
17 if (first == std::string::npos) return "";
18
19 const auto last = s.find_last_not_of(" \t\r\n");
20 return s.substr(first, last - first + 1);
21}

23static std::vector<std::string> SplitString(const std::string& text, char delimiter)
24{
25 std::vector<std::string> tokens;
26 std::stringstream ss(text);
27 std::string item;
28
29 while (std::getline(ss, item, delimiter))
30 {
31 item = Trim(item);
32 if (!item.empty())
33 {
34 tokens.push_back(item);
35 }
36 }
37
38 return tokens;

41static std::unordered_map<std::string, std::string>
42ReadConfigFile(const std::string& filename)
43{
44 std::unordered_map<std::string, std::string> data;
45
46 std::ifstream file(filename);
47 if (!file)
48 {
49 mfem::mfem_error(("Could not open config file: " + filename).c_str());
50 }
51
52 std::string line;
53
54 while (std::getline(file, line))
55 {
56 line = Trim(line);
57
58 if (line.empty()) continue;
59 if (line[0] == '#') continue;
60
61 const auto comment_pos = line.find('#');
62 if (comment_pos != std::string::npos)
63 {
64 line = Trim(line.substr(0, comment_pos));
65 }
66
67 const auto eq_pos = line.find('=');
68 if (eq_pos == std::string::npos) continue;
69
70 std::string key = Trim(line.substr(0, eq_pos));
71 std::string val = Trim(line.substr(eq_pos + 1));
72
73 data[key] = val;
74 }
75
76 return data;
77}

371{
372 const double tolerance = 0.3;
373
374 for (size_t i = 0; i < cfg.cathode_materials.size(); i++)
375 {
376 sim::MaterialType material = cfg.cathode_materials[i];
377 double x = cfg.init_cathode_particles[i];
378
379 double expected_ocv = MaterialProperties::OCV(material, x);
380
381 if (std::abs(cfg.init_BvC - expected_ocv) > tolerance)
382 {
383 std::stringstream ss;
384 ss << "init_BvC = " << cfg.init_BvC
385 << " is far from the OCV curve value for cathode particle "
386 << i << ". For x = " << x
387 << ", suggested init_BvC is about "
388 << expected_ocv << ".";
389
390 mfem::mfem_error(ss.str().c_str());
391 }
