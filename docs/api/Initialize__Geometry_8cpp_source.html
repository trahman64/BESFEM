30static void KeepOnlyConnectedToBoundary_2D(std::vector<uint8_t> &solid, int nx, int ny, bool eight_conn,
31 bool seed_all_boundaries = true, int seed_side = -1)
32{
33 // seed_side: -1 = use all boundaries; 0=left, 1=right, 2=bottom, 3=top
34 auto id = [nx](int i, int j){ return i + nx*j; };
35
36 std::vector<uint8_t> keep(nx*ny, 0);
37 std::queue<std::pair<int,int>> q;
38
39 auto push = [&](int i, int j){
40 if (i < 0 || i >= nx || j < 0 || j >= ny) return;
41 int k = id(i,j);
42 if (!solid[k] || keep[k]) return;
43 keep[k] = 1;
44 q.push({i,j});
45 };
46
47 // seeds
48 if (seed_all_boundaries || seed_side == -1)
49 {
50 for (int i=0;i<nx;i++){ push(i,0); push(i,ny-1); }
51 for (int j=0;j<ny;j++){ push(0,j); push(nx-1,j); }
52 }
53 else
54 {
55 if (seed_side == 0) for (int j=0;j<ny;j++) push(0,j); // left
56 if (seed_side == 1) for (int j=0;j<ny;j++) push(nx-1,j); // right
57 if (seed_side == 2) for (int i=0;i<nx;i++) push(i,0); // bottom
58 if (seed_side == 3) for (int i=0;i<nx;i++) push(i,ny-1); // top
59 }
60
61 const int di4[4] = { 1,-1, 0, 0};
62 const int dj4[4] = { 0, 0, 1,-1};
63 const int di8[8] = { 1,-1, 0, 0, 1, 1,-1,-1};
64 const int dj8[8] = { 0, 0, 1,-1, 1,-1, 1,-1};
65
66 while (!q.empty())
67 {
68 auto [i,j] = q.front(); q.pop();
69 if (!eight_conn)
70 for (int t=0;t<4;t++) push(i+di4[t], j+dj4[t]);
71 else
72 for (int t=0;t<8;t++) push(i+di8[t], j+dj8[t]);
73 }
74
75 // remove islands
76 for (int k=0;k<nx*ny;k++) if (solid[k] && !keep[k]) solid[k] = 0;
77}

79static void KeepOnlyConnectedToBoundary_3D(std::vector<uint8_t> &solid,

102 if (face==1) for (int k=0;k<nz;k++) for (int j=0;j<ny;j++) push(nx-1,j,k); // xmax
103 if (face==2) for (int k=0;k<nz;k++) for (int i=0;i<nx;i++) push(i,0,k); // ymin
104 if (face==3) for (int k=0;k<nz;k++) for (int i=0;i<nx;i++) push(i,ny-1,k); // ymax
105 if (face==4) for (int j=0;j<ny;j++) for (int i=0;i<nx;i++) push(i,j,0); // zmin
106 if (face==5) for (int j=0;j<ny;j++) for (int i=0;i<nx;i++) push(i,j,nz-1); // zmax
107 };
108
109 if (seed_all_boundaries || seed_face == -1)
110 {
111 for (int face=0; face<6; face++) seed_face_fn(face);
112 }
113 else
114 {
115 seed_face_fn(seed_face);
116 }
117
118 // BFS neighbors
119 if (!twenty_six_conn)
120 {
121 const int di[6]={ 1,-1, 0, 0, 0, 0};
122 const int dj[6]={ 0, 0, 1,-1, 0, 0};
123 const int dk[6]={ 0, 0, 0, 0, 1,-1};
124 while(!q.empty())
125 {
126 auto [i,j,k]=q.front(); q.pop();
127 for(int t=0;t<6;t++) push(i+di[t], j+dj[t], k+dk[t]);
128 }
129 }
130 else
131 {
132 while(!q.empty())
133 {
134 auto [i,j,k]=q.front(); q.pop();
135 for(int dk=-1; dk<=1; dk++)
136 for(int dj=-1; dj<=1; dj++)
137 for(int di=-1; di<=1; di++)
138 {
139 if (di==0 && dj==0 && dk==0) continue;
140 push(i+di, j+dj, k+dk);
141 }
142 }
143 }
144
145 // remove islands
146 for (int idx=0; idx<nx*ny*nz; idx++)
147 if (solid[idx] && !keep[idx]) solid[idx] = 0;
148}

365 std::string meshFileStr(meshFile); // Convert to std::string
366 std::string fileExtension = meshFileStr.substr(meshFileStr.find_last_of(".") + 1);

368 if (fileExtension == "tif") {

380 throw std::invalid_argument("Unsupported file format. Only .tif and .mesh are allowed.");

383 // ensure mesh supports non-conforming elements for adaptive refinement
384 globalMesh->EnsureNCMesh(true);

386
387 int e = 0;
388 mfem::Array<int> vert_ids;
389 globalMesh->GetElementVertices(e, vert_ids);
390
391 mfem::Vector v0(globalMesh->GetVertex(vert_ids[0]), globalMesh->SpaceDimension());
392 mfem::Vector v1(globalMesh->GetVertex(vert_ids[1]), globalMesh->SpaceDimension());
393
394
395 double dh1 = v0.DistanceTo(v1);
396 if (mfem::Mpi::WorldRank() == 0) { std::cout << "Element size dh = " << dh1 << std::endl;}
397
398 MPI_Barrier(MPI_COMM_WORLD);

749void Initialize_Geometry::SaveTiffDataToPGM(
750 const std::vector<std::vector<std::vector<int>>> &data,
751 const std::string &filename)
752{
753 if (data.empty() || data[0].empty() || data[0][0].empty()) {
754 std::cerr << "SaveTiffDataToPGM: empty data\n";
755 return;
756 }
757
758 const auto &img = data[0];
759 const int height = static_cast<int>(img.size());
760 const int width = static_cast<int>(img[0].size());
761
762 int max_label = 0;
763 for (const auto &row : img) {
764 for (int label : row) {
765 max_label = std::max(max_label, label);
766 }
767 }
768
769 std::ofstream out(filename, std::ios::binary);
770 if (!out.is_open()) {
771 std::cerr << "Could not open file for writing: " << filename << "\n";
772 return;
773 }
774
775 out << "P5\n" << width << " " << height << "\n255\n";
776
777 for (int j = 0; j < height; ++j) {
778 for (int i = 0; i < width; ++i) {
779 const int label = img[j][i];
780
781 unsigned char val = 0;
782 if (max_label > 0) {
783 val = static_cast<unsigned char>(
784 std::round(255.0 * label / max_label)
785 );
786 }
787
788 out.write(reinterpret_cast<char*>(&val), 1);
789 }
790 }
791
792 out.close();
793
794 if (mfem::Mpi::WorldRank() == 0) {
795 std::cout << "Saved PGM to " << filename
796 << " using labels 0-" << max_label << "\n";
797 }

801void Initialize_Geometry::ComputePDEFilter(mfem::ParGridFunction &dist, mfem::ParGridFunction &filt_gf, int mode)
802
803{
804 MFEM_VERIFY(parallelMesh, "parallelMesh is not initialized.");
805 MFEM_VERIFY(parfespace, "parfespace is not initialized.");
806 MFEM_VERIFY(Vox, "Vox is not initialized (need .tif path + MapGlobalToLocal).");
807 MFEM_VERIFY(dist.ParFESpace() == parfespace.get(), "dist must be on parfespace.");
808 MFEM_VERIFY(filt_gf.ParFESpace() == parfespace.get(), "filt_gf must be on parfespace.");
809 MFEM_VERIFY(parfespace_dg, "parfespace_dg is not initialized.");
810
811 double dx;
812 dx = parallelMesh->GetElementSize(0); // assuming uniform mesh
813
814 MFEM_VERIFY(parallelMesh->Dimension() == 2 || parallelMesh->Dimension() == 3,
815 "ComputePDEFilter: mesh must be 2D or 3D.");
816
817 // TIFF sizes
818 const int nz = (int)tiffData.size();
819 const int ny = (int)tiffData[0].size();
820 const int nx = (int)tiffData[0][0].size();
821
822 const int rank = mfem::Mpi::WorldRank();
823 const bool eight_conn = false; // 2D: 4/8
824 const bool twenty_six = false; // 3D: 6/26 (set true if desired)
825
826 // IMPORTANT: fg must cover the whole volume for 3D
827 std::vector<uint8_t> fg(nx*ny*nz, 0);
828
829 // Boundary Rules: [0] = west, [1] = east, [2] = south, [3] = north, [4] = bottom, [5] = top
830
831 if (rank == 0)
832 {
833 // base solid from TIFF: 1 = white/solid
834 std::vector<uint8_t> solid_base(nx*ny*nz, 0);
835 for (int k=0; k<nz; ++k)
836 for (int j=0; j<ny; ++j)
837 for (int i=0; i<nx; ++i)
838 {
839 const int idx = i + nx*j + nx*ny*k;
840 solid_base[idx] = (tiffData[k][j][i] > 0) ? 1 : 0;
841 }
842
843 if (mode == 0)
844 {
845 // PSI: solid phase, keep only stuff connected to a boundary
846 fg = solid_base;
847
848 if (nz == 1)
849 {
850 KeepOnlyConnectedToBoundary_2D(fg, nx, ny, eight_conn, false, 1); // psi boundary
851 }
852 else
853 {
854 // xmax face in 3D (seed_face=1)
855 KeepOnlyConnectedToBoundary_3D(fg, nx, ny, nz, twenty_six, false, 1);
856 }
857 }
858 else if (mode == 1)
859 {
860 // PSE: electrolyte = NOT solid
861 for (int idx=0; idx<nx*ny*nz; ++idx) fg[idx] = solid_base[idx] ? 0 : 1;
862
863 if (nz == 1)
864 {
865 KeepOnlyConnectedToBoundary_2D(fg, nx, ny, eight_conn, false, 0); // pse boundary
866 // KeepOnlyConnectedToBoundary_2D(fg, nx, ny, eight_conn, true, -1); // all boundaries
867 }
868 else
869 {
870 KeepOnlyConnectedToBoundary_3D(fg, nx, ny, nz, twenty_six, false, 0);
871 }
872 }
873 else
874 {
875 MFEM_ABORT("ComputePDEFilter: mode must be 0 (psi) or 1 (pse).");
876 }
877 }
878
879 // Broadcast full mask to all ranks
880 MPI_Bcast(fg.data(), (int)fg.size(), MPI_UNSIGNED_CHAR, 0, MPI_COMM_WORLD);
881
882 mfem::ParGridFunction ls_coeff_dg(parfespace_dg.get());
883 mfem::ParGridFunction filt_dg(parfespace_dg.get());
884
885 ls_coeff_dg = 0.0;
886 filt_dg = 0.0;
887
888 struct FGCoeffND : public mfem::Coefficient
889 {
890 int nx, ny, nz;
891 int dim;
892 double x0,y0,z0, dxp,dyp,dzp;
893 const std::vector<uint8_t> *fg;
894
895 FGCoeffND(int nx_, int ny_, int nz_, mfem::ParMesh &pmesh, const std::vector<uint8_t> &fg_)
896 : nx(nx_), ny(ny_), nz(nz_), fg(&fg_)
897 {
898 dim = pmesh.Dimension();
899 mfem::Vector bbmin, bbmax;
900 pmesh.GetBoundingBox(bbmin, bbmax);
901
902 x0 = bbmin(0); y0 = bbmin(1);
903 dxp = (bbmax(0) - bbmin(0)) / (nx - 1);
904 dyp = (bbmax(1) - bbmin(1)) / (ny - 1);
905
906 if (dim == 3)
907 {
908 z0 = bbmin(2);
909 dzp = (bbmax(2) - bbmin(2)) / (nz - 1);
910 }
911 else
912 {
913 z0 = 0.0; dzp = 1.0;
914 }
915 }
916
917 double Eval(mfem::ElementTransformation &T, const mfem::IntegrationPoint &ip) override
918 {
919 mfem::Vector X;
920 T.Transform(ip, X);
921
922 int i = (int)std::floor((X(0) - x0)/dxp + 0.5);
923 int j = (int)std::floor((X(1) - y0)/dyp + 0.5);
924 i = std::max(0, std::min(nx-1, i));
925 j = std::max(0, std::min(ny-1, j));
926
927 int k = 0;
928 if (dim == 3)
929 {
930 k = (int)std::floor((X(2) - z0)/dzp + 0.5);
931 k = std::max(0, std::min(nz-1, k));
932 }
933
934 const int idx = i + nx*j + nx*ny*k;
935 return (*fg)[idx] ? +1.0 : -1.0;
936 }
937 };
938
939 FGCoeffND fgcoef(nx, ny, nz, *parallelMesh, fg);
940 ls_coeff_dg.ProjectCoefficient(fgcoef);
941
942
943
944 // ------------------ PDEFilter ------------------
945 const double filter_weight = 3 * dx;
946 mfem::common::PDEFilter filter(*parallelMesh, filter_weight);
947 filter.Filter(ls_coeff_dg, filt_dg);
948
949 for (int i = 0; i < filt_dg.Size(); i++)
950 {
951 filt_dg(i) = 0.5*(filt_dg(i) + 1.0);
952 }
953
954 mfem::GridFunctionCoefficient ls_filt_coeff(&filt_dg);
955
956 filt_gf.ProjectGridFunction(filt_dg);
957}
